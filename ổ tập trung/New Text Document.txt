function getRightPanelFriendName(){
	//v7.3.0 update - Nếu là dạng thẻ svg (mới) thì xử lý lấy attr
	if(q(SELECTOR.InRightPanel.FriendName )){

		let friendNameTag = q(SELECTOR.InRightPanel.FriendName);
		let friendName;

		//Xử lý phân loại thẻ để lấy inner hay aria-label svg tại update v7.3.0

		if(friendNameTag.tagName == "svg"){
			friendName = friendNameTag.getAttribute("aria-label").trim();
		}else{//Dạng cũ, giữ
			friendName = friendNameTag.innerText.trim();
		}

		if(friendName.indexOf(")") > -1){
			friendName = friendName.split("(")[0].trim();
		}
		return friendName;
	}
	return null;
}

	function getServerTimestamp(database) {
		return new Promise((resolve, reject) => {
			const offsetRef = database.ref(".info/serverTimeOffset");
			offsetRef.once('value').then((snapshot) => {
				const offset = snapshot.val();
				if (offset) {
					const serverTime = Date.now() + offset; // Timestamp chính xác từ server
					debug("serverTime lấy được : " + serverTime);
					resolve(serverTime);
				} else {
					debug("Lỗi lấy Servertime (không có giá trị)");
					resolve(false);
				}
			}).catch((error) => {
				debug("Lỗi lấy Servertime : " + error);
				resolve(false);
			});
		});
	}

	function parseDDMMYYYYToValidDateFormat(ddmmyyy) {

    // Tách chuỗi theo dấu '/' dạng dd/mm/yyyy
    const parts = ddmmyyy.split('/');
    
    // Kiểm tra xem định dạng có đúng không
    if (parts.length !== 3) {
        throw new Error("Invalid date format. Expected format: dd/mm/yyyy");
    }
    
    // Lấy ngày, tháng, năm
    const day = parseInt(parts[0], 10);
    const month = parseInt(parts[1], 10) - 1; // Tháng trong JavaScript bắt đầu từ 0
    const year = parseInt(parts[2], 10);
    
    // Tạo đối tượng Date
    const date = new Date(year, month, day);
    
    // Kiểm tra xem ngày có hợp lệ không
    if (date.getFullYear() !== year || date.getMonth() !== month || date.getDate() !== day) {
        throw new Error("Invalid date");
    }
    
    return date;
}

	function startCountDown(){
		debug("Load lại sau : "+refreshMin + " phút");

			var countDownDate = addMinutes(new Date(),refreshMin).getTime();
			 startCountDownTask = setInterval(function() {
			var now = new Date().getTime();
			var distance = countDownDate - now;
			var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
			var seconds = Math.floor((distance % (1000 * 60)) / 1000);
			let time = (minutes < 10 ? "0"+ minutes : minutes) + ":" +  (seconds < 10 ? "0"+ seconds : seconds);

			document.getElementById("timer_id_kbdp").innerHTML = time;
			
			if(document.title.includes("[")){
				document.title = `[${time}] ${document.title.substr(8)}`;
			}else{
				document.title = `[${time}] ${document.title}`;

			}

			if (distance < 0) {
				clearInterval(startCountDownTask);
				document.getElementById("timer_id_kbdp").innerHTML = "Đang tải kết quả mới...";
				setTimeout(toKBDPLink,3000);
			}
		}, 1000);
	}
