// ===== Lấy DOM =====
const mainPanel = document.getElementById('mainPanel');
const stopPanel = document.getElementById('stopPanel');
const progressFill = document.getElementById('progressFill');
const progressText = document.getElementById('progressText');
const percentText = document.getElementById('percentText');
const remainCount = document.getElementById('remainCount');
const sentDisplay = document.getElementById('sentDisplay');
const sentStat = document.getElementById('sentStat');
const skippedStat = document.getElementById('skippedStat');
const targetStat = document.getElementById('targetStat');
const configGender = document.getElementById('configGender');
const configMutual = document.getElementById('configMutual');
const configDelay = document.getElementById('configDelay');
const activityLog = document.getElementById('activityLog');
const liveBadge = document.getElementById('liveBadge');

const stopSent = document.getElementById('stopSent');
const stopSkipped = document.getElementById('stopSkipped');
const stopTarget = document.getElementById('stopTarget');
const stopRemain = document.getElementById('stopRemain');
const stopConfigDisplay = document.getElementById('stopConfigDisplay');

// ===== Hàm cập nhật UI =====
function updateUI() {
  chrome.storage.local.get(['config', 'sentCount', 'skippedCount', 'maxInvites', 'activityLog', 'isStopped'], (data) => {
    const config = data.config || { gender: 'female', mutualFriends: 0, delayMin: 6, delayMax: 10 };
    const sent = data.sentCount || 0;
    const skipped = data.skippedCount || 0;
    const max = data.maxInvites || 50;
    const log = data.activityLog || [];
    const stopped = data.isStopped || false;

    // Cập nhật live panel
    const percent = Math.min(100, Math.round((sent / max) * 100));
    progressFill.style.width = percent + '%';
    progressText.textContent = percent + '%';
    percentText.textContent = percent + '%';
    remainCount.textContent = Math.max(0, max - sent);
    sentDisplay.textContent = sent;
    sentStat.textContent = sent;
    skippedStat.textContent = skipped;
    targetStat.textContent = max;

    const genderLabel = config.gender === 'female' ? 'Nữ' : 'Nam';
    configGender.textContent = genderLabel;
    configMutual.textContent = (config.mutualFriends || 0) + '+';
    configDelay.textContent = (config.delayMin || 6) + '-' + (config.delayMax || 10) + 's';

    // Activity log
    activityLog.innerHTML = '';
    const recent = log.slice(-5).reverse();
    if (recent.length === 0) {
      activityLog.innerHTML = '<div style="color:#757575;">Chưa có hoạt động</div>';
    } else {
      recent.forEach(msg => {
        const div = document.createElement('div');
        div.textContent = msg;
        activityLog.appendChild(div);
      });
    }

    // Stop panel
    stopSent.textContent = sent;
    stopSkipped.textContent = skipped;
    stopTarget.textContent = max;
    stopRemain.textContent = Math.max(0, max - sent);
    stopConfigDisplay.textContent = `Giới tính ${genderLabel}, bạn chung ${config.mutualFriends || 0}+, chờ ${config.delayMin || 6}-${config.delayMax || 10}s`;

    // Chuyển panel nếu dừng
    if (stopped) {
      mainPanel.classList.add('hidden');
      stopPanel.classList.remove('hidden');
      liveBadge.textContent = 'Đã dừng';
      liveBadge.style.background = '#ff5252';
      liveBadge.style.animation = 'none';
    } else {
      mainPanel.classList.remove('hidden');
      stopPanel.classList.add('hidden');
      liveBadge.textContent = 'Live';
      liveBadge.style.background = '#00c853';
      liveBadge.style.animation = 'pulse 1.2s infinite';
    }
  });
}

// ===== Lắng nghe tin nhắn từ content script =====
chrome.runtime.onMessage.addListener((msg) => {
  if (msg.type === 'progress') {
    if (msg.name) {
      chrome.storage.local.get(['activityLog'], (data) => {
        let log = data.activityLog || [];
        log.push('Đã gửi lời mời cho ' + msg.name);
        chrome.storage.local.set({ activityLog: log }, updateUI);
      });
    } else {
      updateUI();
    }
  }
  if (msg.type === 'skipped') {
    chrome.storage.local.get(['skippedCount'], (data) => {
      let skipped = (data.skippedCount || 0) + 1;
      chrome.storage.local.set({ skippedCount: skipped }, updateUI);
    });
  }
  if (msg.action === 'stopped') {
    chrome.storage.local.set({ isStopped: true }, updateUI);
  }
});

// ===== Nút bắt đầu (đã có trong popup, nhưng giờ xử lý tại đây) =====
// Tạo nút "Bắt đầu" mới nếu chưa có
const startBtn = document.createElement('button');
startBtn.textContent = 'Bắt đầu';
startBtn.style.cssText = 'background:#4fc3f7;border:none;color:#0b0e14;padding:8px;border-radius:6px;font-weight:bold;width:100%;margin-top:8px;cursor:pointer;';
startBtn.addEventListener('click', () => {
  // Lấy config từ storage hoặc tạo mới
  chrome.storage.local.get(['config'], (data) => {
    let config = data.config || { gender: 'female', mutualFriends: 0, maxInvites: 50, delayMin: 6, delayMax: 10 };
    chrome.storage.local.set({ 
      config, 
      sentCount: 0, 
      skippedCount: 0, 
      activityLog: [], 
      isStopped: false,
      maxInvites: config.maxInvites 
    });
    chrome.tabs.query({ active: true, currentWindow: true }, (tabs) => {
      if (tabs[0]) {
        chrome.tabs.sendMessage(tabs[0].id, { action: 'start', config });
      }
    });
    updateUI();
  });
});
// Chèn vào đầu mainPanel
mainPanel.insertBefore(startBtn, mainPanel.firstChild);

// ===== Nút dừng =====
document.getElementById('stopBtn').addEventListener('click', () => {
  chrome.tabs.query({ active: true, currentWindow: true }, (tabs) => {
    if (tabs[0]) {
      chrome.tabs.sendMessage(tabs[0].id, { action: 'stop' });
    }
  });
  chrome.storage.local.set({ isStopped: true }, updateUI);
});

// ===== Nút "Đổi câu hình" =====
document.getElementById('configBtn').addEventListener('click', () => {
  // Mở lại popup với cài đặt, nhưng không cần
  // Có thể mở trang cài đặt riêng hoặc reload popup
  chrome.runtime.reload();
});

// ===== Nút "Xong" =====
document.getElementById('doneBtn').addEventListener('click', () => {
  window.close();
});

// ===== Khởi tạo và cập nhật định kỳ =====
updateUI();
setInterval(updateUI, 2000);

// Thông báo đã sẵn sàng
console.log('Popup TOPID Assistant loaded');
