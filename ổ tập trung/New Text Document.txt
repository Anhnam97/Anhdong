const DOC_ID = '26828219736856793';

let isRunning = false;

// === HÀM LOG ===
function log(message, type = 'info') {
    const div = document.createElement('div');
    div.className = 'log-' + type;
    const time = new Date().toLocaleTimeString();
    div.textContent = '[' + time + '] ' + message;
    logBox.appendChild(div);
    logBox.scrollTop = logBox.scrollHeight;
}

// === HÀM TIMER ===
function startTimer() {
    startTime = Date.now();
    if (timerInterval) clearInterval(timerInterval);
    timerInterval = setInterval(function() {
        var elapsed = Math.floor((Date.now() - startTime) / 1000);
        var minutes = String(Math.floor(elapsed / 60)).padStart(2, '0');
        var seconds = String(elapsed % 60).padStart(2, '0');
        runTimeEl.textContent = minutes + ':' + seconds;
    }, 1000);
}

function stopTimer() {
    if (timerInterval) {
        clearInterval(timerInterval);
        timerInterval = null;
    }
}

// === LẤY USER ID ===
function getCurrentUserId() {
    var cookies = document.cookie.split(';');
    for (var i = 0; i < cookies.length; i++) {
        var c = cookies[i].trim();
        if (c.indexOf('c_user=') === 0) {
            return c.split('=')[1];
        }
    }
    return null;
}

// === LẤY DANH SÁCH ID ===
async function getFollowingIds() {
    log('🔄 Đang cuộn load danh sách...', 'info');
    
    for (var i = 0; i < 20; i++) {
        window.scrollBy(0, 2000);
        await sleep(500);
    }
    
    var ids = new Set();
    var links = document.querySelectorAll('a[href*="/user/"], a[href*="profile.php"]');
    
    for (var i = 0; i < links.length; i++) {
        var match = links[i].href.match(/[?&]id=(\d+)/) || links[i].href.match(/\/(\d{7,15})(?:\/|$)/);
        if (match && match[1] && match[1].length >= 7) {
            ids.add(match[1]);
        }
    }
    
    var result = [];
    for (var id of ids) {
        result.push(id);
    }
    
    log('📋 Tìm thấy ' + result.length + ' ID đang theo dõi', 'info');
    return result;
}

function sleep(ms) {
    return new Promise(r => setTimeout(r, ms));
}

// === GỌI API UNFOLLOW ===
async function unfollowAPI(userId, retry = 0) {
    var actorId = getCurrentUserId();
    if (!actorId) {
        log('❌ Không tìm thấy actor_id!', 'error');
        return false;
    }
    
    try {
        var variables = {
            action_render_location: "WWW_COMET_FRIEND_MENU",
            input: {
                attribution_id_v2: "ProfileCometCollectionRoot.react,comet.profile.collection.following,via_cold_start,1783844581372,740548,,,",
                is_tracking_encrypted: false,
                subscribe_location: "PROFILE",
                tracking: null,
                unsubscribee_id: userId,
                actor_id: actorId,
                client_mutation_id: "1"
            },
            scale: 1
        };
        
        var body = new URLSearchParams();
        body.append('fb_dtsg', YOUR_TOKEN);
        body.append('doc_id', DOC_ID);
        body.append('variables', JSON.stringify(variables));
        
        var res = await fetch('https://www.facebook.com/api/graphql/', {
            method: 'POST',
            credentials: 'include',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded',
                'Accept': 'application/json',
                'X-Requested-With': 'XMLHttpRequest'
            },
            body: body.toString()
        });
        
        var data = await res.json();
        
        if (data.data && (data.data.unsubscribe || data.data.success)) {
            return true;
        }
        
        if (data.errors) {
            var errorMsg = data.errors[0]?.message || 'Unknown error';
            // Nếu lỗi token hết hạn
            if (errorMsg.indexOf('token') !== -1 || errorMsg.indexOf('expired') !== -1) {
                log('⚠️ Token hết hạn! Cần lấy token mới.', 'error');
            }
            return false;
        }
        
        return false;
        
    } catch (e) {
        log('❌ Lỗi: ' + e.message, 'error');
        return false;
    }
}

// === HÀM CHÍNH CHẠY UNFOLLOW ===
async function runUnfollow() {
    const maxCount = parseInt(maxCountInput.value) || 30;
    const delay = parseFloat(delayTimeInput.value) || 2;
    const maxRetry = parseInt(retryCountInput.value) || 2;
    
    // Reset
    unfollowedCount = 0;
    scannedCount = 0;
    currentIndex = 0;
    shouldStop = false;
    isPaused = false;
    unfollowedCountEl.textContent = '0';
    scannedCountEl.textContent = '0';
    progressFill.style.width = '0%';
    
    log('📌 Cấu hình: Số lượng=' + maxCount + ', Delay=' + delay + 's, Thử lại=' + maxRetry + ' lần', 'info');
    
    // Lấy actor_id
    currentUserId = getCurrentUserId();
    if (!currentUserId) {
        log('❌ Không tìm thấy user ID! F5 refresh rồi thử lại.', 'error');
        statusEl.textContent = '❌ Lỗi';
        isRunning = false;
        updateButtons();
        return;
    }
    
    // Lấy danh sách ID
    statusEl.textContent = '🔄 Đang quét ID...';
    allIds = await getFollowingIds();
    
    if (allIds.length === 0) {
        log('❌ Không tìm thấy ID nào! Vào tab "Đang theo dõi"', 'error');
        statusEl.textContent = '❌ Không có ID';
        isRunning = false;
        updateButtons();
        return;
    }
    
    totalIds = allIds.length;
    if (allIds.length > maxCount) {
        allIds = allIds.slice(0, maxCount);
        totalIds = allIds.length;
        log('📌 Giới hạn ' + maxCount + ' tài khoản', 'info');
    }
    
    totalIdsEl.textContent = totalIds;
    statusEl.textContent = '🔄 Đang hủy...';
    startTimer();
    
    // Vòng lặp chính
    for (var i = 0; i < allIds.length; i++) {
        // Kiểm tra dừng
        if (shouldStop) {
            statusEl.textContent = '⏹ Đã dừng';
            log('⏹ Đã dừng bởi người dùng', 'warning');
            break;
        }
        
        // Kiểm tra tạm dừng
        while (isPaused) {
            statusEl.textContent = '⏸ Đang tạm dừng';
            await sleep(500);
            if (shouldStop) break;
        }
        if (shouldStop) break;
        
        currentIndex = i;
        var userId = allIds[i];
        
        currentIdEl.textContent = userId;
        scannedCount = i + 1;
        scannedCountEl.textContent = scannedCount;
        
        // Cập nhật progress
        var progress = ((i + 1) / totalIds) * 100;
        progressFill.style.width = progress + '%';
        
        log('🔄 [' + (i+1) + '/' + totalIds + '] Đang hủy ID: ' + userId, 'info');
        
        // Gọi API với số lần thử lại
        var success = false;
        for (var retry = 0; retry <= maxRetry; retry++) {
            if (retry > 0) {
                log('🔄 Thử lại lần ' + retry + ' cho ID: ' + userId, 'warning');
                await sleep(delay * 1000);
            }
            success = await unfollowAPI(userId);
            if (success) break;
        }
        
        if (success) {
            unfollowedCount++;
            unfollowedCountEl.textContent = unfollowedCount;
            log('✅ #' + unfollowedCount + ' - Đã hủy ID: ' + userId, 'success');
        } else {
            log('❌ Không hủy được ID: ' + userId + ' (đã thử ' + (maxRetry+1) + ' lần)', 'error');
        }
        
        // Delay giữa các lần
        await sleep(delay * 1000);
    }
    
    // Kết thúc
    stopTimer();
    if (!shouldStop && !isPaused) {
        statusEl.textContent = '✅ Hoàn tất';
        log('🎉 HOÀN TẤT! Đã hủy ' + unfollowedCount + '/' + totalIds + ' tài khoản', 'success');
        progressFill.style.width = '100%';
    }
    
    isRunning = false;
    updateButtons();
}

    btnStop.disabled = !isRunning;
}
