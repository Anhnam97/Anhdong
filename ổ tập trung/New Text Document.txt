export function sendToContentScript(event, data) {
  return new Promise((resolve, reject) => {
    let uuid = Math.random().toString(36); // uuid to distinguish events
    let listenerKey = "fbamf-contentscript-sendto-pagescript" + uuid;
    window.addEventListener(listenerKey, (evt) => resolve(evt.detail.data), {
      once: true,
    });
    window.dispatchEvent(
      new CustomEvent("fbamf-pagescript-sendto-contentscript", {
        detail: { event, data, uuid },
      })
    );
  });
}

export function runInContentScript(fnPath, params) {
  // WARNING: can only transfer serializable data
  return sendToContentScript("fbamf-runInContentScript", { fnPath, params });
}

export function runInBackground(fnPath, params) {
  return sendToContentScript("fbamf-runInBackground", { fnPath, params });
}

export function runFunc(fnPath = "", params = [], global = chrome) {
  return new Promise((resolve) => {
    let fn = fnPath?.startsWith("chrome") ? chrome : global;
    fnPath.split(".").forEach((part) => {
      fn = fn?.[part] || fn;
    });

    let hasCallback = false;
    let _params = params.map((p) => {
      if (p === "callback") {
        hasCallback = true;
        return resolve;
      }
      return p;
    });

    if (typeof fn !== "function") return resolve(null);

    try {
      let res = fn(..._params);

      if (!hasCallback) {
        if (typeof res?.then === "function") {
          res.then?.(resolve);
        } else {
          resolve(res);
        }
      }
    } catch (e) {
      console.log("ERROR runFunc: ", e);
      resolve(null);
    }
  });
}

const cache_key = "fb_auto_make_friends_cache";
export function getCache(key, defaultValue = null) {
  const cached = parseSafe(localStorage.getItem(cache_key) || "{}") || {};
  if (key) return cached[key] || defaultValue;
  return cached;
}
export function setCache(key, value) {
  const current = getCache();
  current[key] = value;
  localStorage.setItem(cache_key, JSON.stringify(current));
  return current;
}

export function deleteCache(key) {
  const current = getCache();
  delete current[key];
  localStorage.setItem(cache_key, JSON.stringify(current));
  return current;
}

export async function sleep(
  ms,
  stepCallback,
  step = 1000,
  checkStop = () => false
) {
  if (!stepCallback) return _sleep(ms);

  for (let i = 0; i < ms; i += step) {
    if (checkStop()) return false;
    stepCallback(ms - i);
    await _sleep(step);
  }
  return true;
}

function _sleep(ms) {
  return new Promise((resolve) => setTimeout(() => resolve(true), ms));
}

function escapeHtml(value = "") {
  return String(value)
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#039;");
}

export async function fetchGraphQl(
  params = {},
  url = "",
  autoAddParams = true,
  customParams = {},
  customHeaders = {}
) {
  let query = "";
  const fb_dtsg = await getFbDtsg();
  if (!fb_dtsg) {
    console.error("Missing fb_dtsg token");
    return null;
  }

  if (typeof params === "string")
    query = "&q=" + encodeURIComponent(params) + "&fb_dtsg=" + fb_dtsg;
  else
    query = wrapGraphQlParams({
      ...(autoAddParams
        ? {
            dpr: 1,
            __a: 1,
            __aaid: 0,
            __ccg: "GOOD",
            server_timestamps: true,
            fb_dtsg: fb_dtsg,
          }
        : {}),
      ...customParams,
      ...params,
    });

  const graphQlUrl = url || `${window.location.origin}/api/graphql/`;
  const res = await fetch(graphQlUrl, {
    body: query,
    method: "POST",
    headers: {
      "Content-Type": "application/x-www-form-urlencoded",
      ...customHeaders,
    },
    credentials: "include",
  });

  return res;
}

export function wrapGraphQlParams(params = {}) {
  const formBody = [];
  for (const property in params) {
    const encodedKey = encodeURIComponent(property);
    const value =
      typeof params[property] === "string"
        ? params[property]
        : JSON.stringify(params[property]);
    const encodedValue = encodeURIComponent(value);
    formBody.push(encodedKey + "=" + encodedValue);
  }
  return formBody.join("&");
}

export function parseSafe(str, defaultValue = {}) {
  try {
    if (str?.startsWith("for (;;);")) str = str.replace("for (;;);", "");
    return JSON.parse(str);
  } catch (e) {
    console.log("Cannot parse JSON", e, str);
    return defaultValue;
  }
}

export function waitForElements(selector) {
  return new Promise((resolve, reject) => {
    onElementsAdded(selector, resolve, true);
  });
}

export function onElementsAdded(selector, callback, once) {
  let nodes = document.querySelectorAll(selector);
  if (nodes?.length) {
    callback(nodes);
    if (once) return;
  }

  const observer = new MutationObserver((mutations) => {
    mutations.forEach((mutation) => {
      if (!mutation.addedNodes) return;

      for (let node of mutation.addedNodes) {
        if (node.nodeType != 1) continue; // only process Node.ELEMENT_NODE

        let n = node.matches(selector)
          ? [node]
          : Array.from(node.querySelectorAll(selector));

        if (n?.length) {
          callback(n);
          if (once) observer.disconnect();
        }
      }
    });
  });

  observer.observe(document, {
    childList: true,
    subtree: true,
    attributes: false,
    characterData: false,
  });

  // return disconnect function
  return () => observer.disconnect();
}

export function closest(element, selector) {
  let el = element;
  while (el !== null) {
    if (el.matches(selector)) return el;

    let found = el.querySelector(selector);
    if (found) return found;

    el = el.parentElement;
  }
  return el;
}

export function timeToNow(time) {
  const now = Date.now();
  const diff = now - time;
  return formatTimeDiff(diff);
}

export function formatTimeDiff(diff) {
  // less than minute
  if (diff < 60 * 1000)
    return `${Math.floor(diff / 1000)}${trans({ vi: " giây", en: " sec" })}`;
  // less than hour
  if (diff < 60 * 60 * 1000)
    return `${Math.floor(diff / (60 * 1000))}${trans({
      vi: " phút",
      en: " min",
    })}`;
  // less than day
  if (diff < 24 * 60 * 60 * 1000)
    return `${Math.floor(diff / (60 * 60 * 1000))}${trans({
      vi: " giờ",
      en: " hour",
    })}`;
  // less than month
  if (diff < 30 * 24 * 60 * 60 * 1000)
    return `${Math.floor(diff / (24 * 60 * 60 * 1000))}${trans({
      vi: " ngày",
      en: " day",
    })}`;
  // less than year
  if (diff < 365 * 24 * 60 * 60 * 1000)
    return `${Math.floor(diff / (30 * 24 * 60 * 60 * 1000))}${trans({
      vi: " tháng",
      en: " month",
    })}`;
  // more than year
  return `${Math.floor(diff / (365 * 24 * 60 * 60 * 1000))}${trans({
    vi: " năm",
    en: " year",
  })}`;
}
