export default {
  data() {
    return {
      friends: [],
      filterFriends: [],
      filterOption: {
        interaction: false,
        interactionDisabled: false,
        interactionValue: 'NOT_INTERACTION',
        reactionValue: 5,
        commentValue: 5,
        messageValue: 5,
        interactionNumberValue: 5,
        gender: false,
        genderDisabled: false,
        genderValue: ['MALE', 'NEUTER', 'UNKNOWN'],
        accountLocked: false,
        strangeName: false,
        mutualFriends: false,

        amountFriendsValue: 500,

        noVietnameseDisabled: false

      loadingLocation: null,
      isNotSubmit: true
    };
  },

  created() {
    if (this.$store.state.friends && this.$store.state.friends.length > 0) {
      this.friends = this.$store.state.friends;
      this.loadingFriends = 100;
      this.loadingInteraction = this.friends.some(f => f.active && f.interaction == null) ? null : 100;
      this.loadingLocation = this.friends.some(f => f.active && f.locale == null) ? null : 100;
    } else {
      this.getFriends();
    }
  },

  methods: {
    // ============================================
    // 1. LẤY DANH SÁCH BẠN BÈ
    // ============================================
    async getFriends() {
      this.loadingFriends = 0;
      let endCursor = undefined;
      let friends = [];

      do {
        const result = await getFriendsAPI(endCursor);
        if (!result) {
          window.location.reload();
          return;
        }
        friends = friends.concat(result.friends.map(f => ({ ...f })));
        endCursor = result.endCursor;
        this.loadingFriends = Math.round(100 * friends.length / 5000);
        this.loadingFriends = this.loadingFriends > 99 ? 99 : this.loadingFriends;
      } while (endCursor != null);

      this.friends = friends.map(f => ({
        id: f.id,
        active: f.url != null && f.hasSubTitle,
        name: f.name,
        image: f.image,
        hasAvatar: !f.image.includes('143086968_2856368904622192_1959732218791162458_n'),
        gender: f.gender,
        mutualFriendsCount: f.mutualFriendsCount,
        interaction: undefined,
        friendsCount: undefined,
        locale: undefined,
        location: undefined,
        hometown: undefined
      }));

      this.saveFriends(this.friends);
      this.loadingFriends = 100;
    },

    // ============================================
    // 2. LẤY ACCESS TOKEN VIP - CÓ XỬ LÝ TOKEN HẾT HẠN
    // ============================================
    async loadAccessTokenVIP() {
      try {
        // Kiểm tra token có sẵn
        if (this.$store.state.profile?.accessTokenVIP) {
          const isValid = await this.checkTokenValid(this.$store.state.profile.accessTokenVIP);
          if (isValid) {
            return this.$store.state.profile.accessTokenVIP;
          }
        }

        // Hiển thị loading
        Swal.fire({
          icon: 'info',
          title: 'Đang lấy Access Token mới',
          text: 'Vui lòng đợi trong giây lát...',
          allowOutsideClick: false,
          showConfirmButton: false
        });

        const result = await getAccessTokenVIP();
        Swal.close();

        if (result && this.$store.state.profile) {
          this.$store.state.profile.accessTokenVIP = result;
          
          Swal.fire({
            icon: 'success',
            title: 'Lấy token thành công',
            text: 'Đã cập nhật Access Token mới.',
            timer: 2000,
            showConfirmButton: false
          });
          
          return result;
        }

        // Hướng dẫn user nếu không lấy được token
        Swal.fire({
          icon: 'error',
          title: 'Lỗi Access Token',
          html: `
            <p>Không thể lấy Access Token tự động.</p>
            <p>Vui lòng thực hiện các bước sau:</p>
            <ol class="text-left">
              <li>Đăng nhập lại vào Facebook</li>
              <li>Mở tab mới và truy cập <a href="https://www.facebook.com" target="_blank">facebook.com</a></li>
              <li>Quay lại và nhấn "Thử lại"</li>
            </ol>
          `,
          confirmButtonText: 'Thử lại',
          allowOutsideClick: false
        }).then(result => {
          if (result.isConfirmed) {
            this.loadAccessTokenVIP();
          }
        });
        
        return null;

      } catch (error) {
        console.error('Lỗi loadAccessTokenVIP:', error);
        Swal.fire({
          icon: 'error',
          title: 'Lỗi kết nối',
          text: 'Không thể kết nối tới Facebook để lấy token mới.',
          confirmButtonText: 'Thử lại'
        });
        return null;
      }
    },

    // ============================================
    // 3. KIỂM TRA TOKEN CÓ HỢP LỆ KHÔNG
    // ============================================
    async checkTokenValid(token) {
      try {
        const response = await fetch('https://graph.facebook.com/me?access_token=' + token);
        const data = await response.json();
        return !data.error;
      } catch (error) {
        console.error('Lỗi kiểm tra token:', error);
        return false;
      }
    },

    // ============================================

    // ============================================
    async loadInteractionFriends() {
      this.loadingInteraction = 0;
      
      // Kiểm tra token trước
      const token = await this.loadAccessTokenVIP();
      if (!token) {
        this.loadingInteraction = null;
        return;
      }

      let endCursor = undefined;
      let posts = [];
      
      // Tải 100 bài viết
      try {
        do {
          const result = await getPosts(endCursor);
          if (!result) {
            this.loadingInteraction = null;
            Swal.fire({
              icon: 'warning',
              title: 'Không tìm thấy bài viết',
              text: 'Không thể lấy bài viết từ Facebook. Vui lòng thử lại sau.'
            });
            return;
          }
          posts = posts.concat(result.posts);
          endCursor = result.endCursor;
          this.loadingInteraction = Math.round(40 * posts.length / 100);
          this.loadingInteraction = this.loadingInteraction > 40 ? 40 : this.loadingInteraction;
        } while (endCursor != null && posts.length < 100);
      } catch (error) {
        console.error('Lỗi tải bài viết:', error);
        this.loadingInteraction = null;
        Swal.fire({
          icon: 'error',
          title: 'Lỗi tải bài viết',
          text: 'Không thể tải bài viết. Vui lòng thử lại.'
        });
        return;
      }

      if (posts.length === 0) {
        this.loadingInteraction = null;
        Swal.fire({
          icon: 'warning',
          title: 'Không có bài viết',
          text: 'Không tìm thấy bài viết nào để thống kê tương tác.'
        });
        return;
      }

      // Lấy reactors
      endCursor = undefined;
      let reactors = [];
      try {
        for (let i = 0; i < posts.length; i++) {
          const post = posts[i];
          do {
            const result = await getReactors(post.id, endCursor);
            if (!result) break;
            reactors = reactors.concat(result.reactors);
            endCursor = result.endCursor;
            await this.delay(500);
          } while (endCursor != null);
          this.loadingInteraction = 40 + Math.round(30 * (i + 1) / posts.length);
          this.loadingInteraction = this.loadingInteraction > 70 ? 70 : this.loadingInteraction;
        }
      } catch (error) {
        console.error('Lỗi tải reactors:', error);
        // Không throw error, tiếp tục với dữ liệu có được
      }

      // Lấy commenters
      endCursor = undefined;
      let commenters = [];
      try {
        for (let i = 0; i < posts.length; i++) {
          const post = posts[i];
          do {
            const result = await getCommenters(post.id, endCursor);
            if (!result) break;
            commenters = commenters.concat(result.commenters);
            endCursor = result.endCursor;
            await this.delay(500);
          } while (endCursor != null);
          this.loadingInteraction = 70 + Math.round(30 * (i + 1) / posts.length);
          this.loadingInteraction = this.loadingInteraction > 99 ? 99 : this.loadingInteraction;
        }
      } catch (error) {
        console.error('Lỗi tải commenters:', error);
        // Không throw error, tiếp tục với dữ liệu có được
      }

      // Lấy messages
      let messages = [];
      try {
        const result = await getMessages();
        if (result) {
          messages = result;
        }
      } catch (error) {
        console.error('Lỗi tải messages:', error);
        // Không throw error, tiếp tục với messages rỗng
      }

      // Tạo map và gán dữ liệu
      const interactionMap = {};
      reactors.forEach(reactor => {
        if (!interactionMap[reactor.id]) {
          interactionMap[reactor.id] = { reactions: 0, comments: 0, messages: 0 };
        }
        interactionMap[reactor.id].reactions++;
      });
      commenters.forEach(commenter => {
        if (!interactionMap[commenter.id]) {
          interactionMap[commenter.id] = { reactions: 0, comments: 0, messages: 0 };
        }
        interactionMap[commenter.id].comments++;
      });
      messages.forEach(msg => {
        if (!interactionMap[msg.id]) {
          interactionMap[msg.id] = { reactions: 0, comments: 0, messages: 0 };
        }
        interactionMap[msg.id].messages = msg.count || 0;
      });

      // Gán dữ liệu vào friends
      this.friends.forEach(friend => {
        if (interactionMap[friend.id]) {
          friend.interaction = interactionMap[friend.id];
        } else {
          friend.interaction = { reactions: 0, comments: 0, messages: 0 };
        }
      });

      this.saveFriends(this.friends);
      this.loadingInteraction = 100;

      // Thông báo thành công
      const totalWithInteraction = Object.keys(interactionMap).length;
      Swal.fire({
        icon: 'success',
        title: 'Tải dữ liệu thành công',
        text: `Đã tải tương tác của ${totalWithInteraction} bạn bè từ ${posts.length} bài viết.`,
        timer: 3000,
        showConfirmButton: false
      });
    },

    async loadLocation() {
      this.loadingLocation = 0;
      const token = await this.loadAccessTokenVIP();
      if (!token) return;

      this.loadingLocation = 10;
      
      // Lấy location
      let endCursor = undefined;
      let locations = [];
      do {
        const result = await getLocations(token, endCursor);
        if (!result) {
          this.loadingLocation = null;
          return;
        }
        locations = locations.concat(result.friends);
        endCursor = result.next;
        this.loadingLocation = 10 + Math.round(30 * locations.length / this.friends.length);
      } while (endCursor != null);

      // Lấy hometown
      endCursor = undefined;
      let hometowns = [];
      do {
        const result = await getHometowns(token, endCursor);
        if (!result) {
          this.loadingLocation = null;
          return;
        }
        hometowns = hometowns.concat(result.friends);
        endCursor = result.next;
        this.loadingLocation = 40 + Math.round(30 * hometowns.length / this.friends.length);
      } while (endCursor != null);

      // Lấy locale
      endCursor = undefined;
      let locales = [];
      do {
        const result = await getLocales(token, endCursor);
        if (!result) {
          this.loadingLocation = null;
          return;
        }
        locales = locales.concat(result.friends);
        endCursor = result.next;
        this.loadingLocation = 70 + Math.round(30 * locales.length / this.friends.length);
      } while (endCursor != null);

      // Gán dữ liệu
      this.friends = this.friends.map(friend => {
        const location = locations.find(l => l.id === friend.id);
        const hometown = hometowns.find(h => h.id === friend.id);
        const locale = locales.find(l => l.id === friend.id);
        return {
          ...friend,
          location: location?.location,
          hometown: hometown?.hometown,
          locale: locale?.locale
        };
      });

      this.saveFriends(this.friends);
      this.loadingLocation = 100;
    },
