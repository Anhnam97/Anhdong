  function getSuggestTargets() {
    const candidates = [];
    const seenRoots = new WeakSet();

    for (const target of Array.from(
      document.querySelectorAll(
        [
          '[aria-label="ThÃªm báº¡n bÃ¨"]',
          '[aria-label*="Káº¿t báº¡n vá»›i"]',
          '[aria-label*="ThÃªm báº¡n"]',
        ].join(",")
      )
    )) {
      if (target.id === btnId || target.offsetParent === null) continue;
      const button = getAddFriendButton(target, true);
      if (!button || button.offsetParent === null) continue;
      const root = getCandidateRoot(button);
      if (root && seenRoots.has(root)) continue;
      if (root) seenRoots.add(root);
      candidates.push(button);
    }

    return candidates;
  }

  function getCandidateRoot(target) {
    const structured =
      target?.closest?.('[role="article"]') || target?.closest?.('[aria-posinset]');
    if (structured) return structured;

    let node = target;
    for (let i = 0; i < 8 && node?.parentElement; i++) {
      node = node.parentElement;
      if (isCandidateRoot(node)) {
        return node;
      }
    }

    return (
      target?.closest?.('[data-visualcompletion="ignore-dynamic"]') ||
      target?.parentElement ||
      target
    );
  }

  function isCandidateRoot(node) {
    const text = cleanCandidateText(node?.innerText || node?.textContent || "");
    if (!text.includes("ThÃªm báº¡n bÃ¨")) return false;
    if (!/(Gá»¡|báº¡n chung|ngÆ°á»i theo dÃµi|ÄÃ£ gá»­i lá»i má»i)/i.test(text)) return false;

    const hasName =
      Array.from(node.querySelectorAll?.('a[href], img[alt]') || []).some(
        (element) =>
          cleanCandidateName(
            element.getAttribute("aria-label") ||
              element.getAttribute("alt") ||
              element.textContent
          )
      ) ||
      text
        .split("\n")
        .some((line) => Boolean(cleanCandidateName(line)));

    return hasName;
  }

  function getProfileHref(target, suggestMode) {
    if (!target) return null;
    if (suggestMode && target.href) return target.href;
    if (!suggestMode) return closest(target, "a[href]")?.href;

    const root = getCandidateRoot(target);
    const profileLink =
      target.closest?.('a[role="link"][href]') ||
      root?.querySelector?.('a[role="link"][href]') ||
      root?.querySelector?.('a[href]');
    return profileLink?.href || null;
  }

  function getCandidateKey(target, suggestMode, href = "") {
    if (href) return normalizeCandidateHref(href);
    const root = getCandidateRoot(target);
    const name = getCandidateName(target);
    const rootText = cleanCandidateText(root?.innerText || root?.textContent || "");
    return `${suggestMode ? "suggest" : "friends"}:${name}:${rootText.slice(0, 120)}`;
  }

  function normalizeCandidateHref(href = "") {
    try {
      const url = new URL(href, window.location.origin);
      url.search = "";
      url.hash = "";
      return url.href.replace(/\/$/, "");
    } catch (e) {
      return String(href).split("?")[0].replace(/\/$/, "");
    }
  }

  function getCandidateName(target) {

    const ariaName = cleanCandidateName(
      ariaLabel
        .replace(/^Káº¿t báº¡n vá»›i\s+/i, "")

    for (const root of getCandidateRoots(target)) {
      const links = Array.from(
        root?.querySelectorAll?.('a[role="link"][href], a[href]') || []

      for (const link of links) {
        const name = cleanCandidateName(
          link.getAttribute("aria-label") || link.textContent

      }

    for (const root of getCandidateRoots(target)) {
      const images = Array.from(
        root?.querySelectorAll?.('img[alt], image[aria-label]') || []
      );
      for (const image of images) {
        const name = cleanCandidateName(
          image.getAttribute("alt") || image.getAttribute("aria-label")
        );
        if (name) return name;
      }
    }

    for (const root of getCandidateRoots(target)) {
      const lines = (root?.innerText || root?.textContent || "")
        .split("\n")
        .map((line) => cleanCandidateName(line))
        .filter(Boolean);
      if (lines.length) return lines[0];
    }

    return unknownCandidateName;
  }

  function getCandidateRoots(target) {
    const roots = [];
    const add = (node) => {
      if (node && !roots.includes(node)) roots.push(node);
    };

    add(getCandidateRoot(target));
    let node = target;
    for (let i = 0; i < 6 && node?.parentElement; i++) {
      node = node.parentElement;
      add(node);
    }

    return roots;
  }

  function cleanCandidateText(value = "") {
    return String(value)
      .replace(/\u200c|\u200d/g, "")
      .replace(/\s+/g, " ")
      .trim();

  function formatCandidateAction(action, name, detail = "") {
    const separator = /(cho|qua|gá»¡)$/i.test(action.trim()) ? " " : ": ";
    return `${action}${separator}${name || unknownCandidateName}${
      detail ? `: ${detail}` : ""

  }

  function updateCandidateName(candidate, name) {
    const cleanName = cleanCandidateName(name);
    if (!cleanName) return candidate.name;
    if (!candidate.name || candidate.name === unknownCandidateName) {
      candidate.name = cleanName;
    }
    return candidate.name;
  }
