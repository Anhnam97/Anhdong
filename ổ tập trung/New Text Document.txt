function getRightPanelFriendName(){
	if(q(SELECTOR.InRightPanel.FriendName )){

		let friendNameTag = q(SELECTOR.InRightPanel.FriendName);
		let friendName;

		if(friendNameTag.tagName == "svg"){
			friendName = friendNameTag.getAttribute("aria-label").trim();
		}else{
			friendName = friendNameTag.innerText.trim();
		}

		if(friendName.indexOf(")") > -1){
			friendName = friendName.split("(")[0].trim();
		}
		return friendName;
	}
	return null;
}

function getServerTimestamp(database) {
	return new Promise((resolve, reject) => {
		const offsetRef = database.ref(".info/serverTimeOffset");
		offsetRef.once('value').then((snapshot) => {
			const offset = snapshot.val();
			if (offset) {
				const serverTime = Date.now() + offset;
				debug("serverTime lấy được : " + serverTime);
				resolve(serverTime);
			} else {
				debug("Lỗi lấy Servertime (không có giá trị)");
				resolve(false);
			}
		}).catch((error) => {
			debug("Lỗi lấy Servertime : " + error);
			resolve(false);
		});

function parseDDMMYYYYToValidDateFormat(ddmmyyy) {
    const parts = ddmmyyy.split('/');
    if (parts.length !== 3) {
        throw new Error("Invalid date format. Expected format: dd/mm/yyyy");
    }
    const day = parseInt(parts[0], 10);
    const month = parseInt(parts[1], 10) - 1;
    const year = parseInt(parts[2], 10);
    const date = new Date(year, month, day);
    if (date.getFullYear() !== year || date.getMonth() !== month || date.getDate() !== day) {
        throw new Error("Invalid date");
    }
    return date;
}
