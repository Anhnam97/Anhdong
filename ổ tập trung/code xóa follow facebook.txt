const { logger, getCookies } = require('./utils');

class BlockChecker {
  constructor() {

  }

  async checkBlockStatus() {
    this.checkCount++;
    logger.info('🔍 Kiem tra chan lan #' + this.checkCount + '...');

    try {
      const response = await axios.get('https://www.facebook.com/me', {
        params: { fields: 'id,name' },
        headers: {
          'Cookie': this.cookies.cookie,
          'User-Agent': this.cookies.user_agent
        },
        timeout: 10000
      });

      if (response.data && response.data.id) {

        return false;

      return false;
    } catch (error) {

        if (status === 403 || status === 401) {

          this.blockStartTime = Date.now();

      logger.warn('⚠️ Loi kiem tra: ' + error.message);
      return false;

    const isComplete = elapsedMinutes >= cooldownMinutes;

      this.blockStartTime = null;

const { logger, getCookies, getSettings, sleep } = require('./utils');

class AutoFriend {
  constructor() {
    this.cookies = getCookies();
    this.settings = getSettings();
    this.totalFriends = 0;
    this.currentSessionFriends = 0;
    this.isRunning = false;
    this.failedRequests = 0;
    this.successRequests = 0;
  }

  async sendFriendRequest(userId, userName = 'Unknown') {
    try {
      const response = await axios.post(
        'https://www.facebook.com/' + userId + '/friends',
        {},
        {
          headers: {
            'Cookie': this.cookies.cookie,
            'User-Agent': this.cookies.user_agent,
            'Content-Type': 'application/json'
          },
          timeout: 15000
        }
      );
      if (response.status === 200) {
        this.totalFriends++;
        this.currentSessionFriends++;
        this.successRequests++;
        logger.info('✅ Da gui loi moi den ' + userName);
        return true;
      }
      this.failedRequests++;
      return false;
    } catch (error) {
      this.failedRequests++;
      logger.error('❌ Loi gui loi moi: ' + error.message);
      return false;
    }
  }

  async getCurrentFriends() {
    try {
      const response = await axios.get('https://www.facebook.com/me/friends', {
        params: { limit: 100, fields: 'id,name' },
        headers: {
          'Cookie': this.cookies.cookie,
          'User-Agent': this.cookies.user_agent
        },
        timeout: 10000
      });
      return response.data.data || [];
    } catch (error) {
      logger.error('❌ Lay danh sach ban be that bai: ' + error.message);
      return [];
    }
  }

  async getSuggestedUsers() {
    try {
      const currentFriends = await this.getCurrentFriends();
      const friendIds = new Set(currentFriends.map(f => f.id));
      const response = await axios.get('https://www.facebook.com/me/friends', {
        params: { limit: 50, fields: 'id,name' },
        headers: {
          'Cookie': this.cookies.cookie,
          'User-Agent': this.cookies.user_agent
        },
        timeout: 10000
      });
      const suggested = response.data.data || [];
      const filtered = suggested.filter(user => !friendIds.has(user.id));
      logger.info('📋 Tim thay ' + filtered.length + ' nguoi dung de xuat');
      return filtered;
    } catch (error) {
      logger.error('❌ Lay danh sach de xuat that bai: ' + error.message);
      return [];
    }
  }

  async runAutoFriend(targetCount = 5) {
    if (this.isRunning) {
      logger.warn('⚠️ Qua trinh auto friend dang chay');
      return false;
    }
    this.isRunning = true;
    this.currentSessionFriends = 0;
    this.failedRequests = 0;
    this.successRequests = 0;
    try {
      const users = await this.getSuggestedUsers();
      if (users.length === 0) {
        logger.warn('⚠️ Khong tim thay nguoi dung de ket ban');
        this.isRunning = false;
        return false;
      }
      let successCount = 0;
      const target = Math.min(targetCount, users.length);
      for (let i = 0; i < users.length && successCount < target; i++) {
        const user = users[i];
        const result = await this.sendFriendRequest(user.id, user.name);
        if (result) successCount++;
        const delay = this.settings.delay_between_requests || 3000;

      return false;

class ScheduleManager {
  constructor() {
    this.settings = getSettings();
    this.workStartTime = null;
    this.workEndTime = null;
    this.isResting = false;
    this.cycleCount = 0;
    this.totalWorkTime = 0;
    this.totalRestTime = 0;
  }

  startWorkCycle() {
    this.cycleCount++;
    this.workStartTime = Date.now();
    const workDuration = (this.settings.work_duration || 60) * 60 * 1000;
    this.workEndTime = this.workStartTime + workDuration;
    this.isResting = false;
    const minutes = Math.floor(workDuration / 60000);
    logger.info('🚀 Bat dau chu ky lam viec #' + this.cycleCount + ' (' + minutes + ' phut)');
  }

  async startRestCycle() {
    this.isResting = true;
    const restDuration = (this.settings.rest_duration || 10) * 60 * 1000;
    this.totalRestTime += restDuration;
    const minutes = Math.floor(restDuration / 60000);
    logger.info('☕ Bat dau nghi ngoi (' + minutes + ' phut)');
    await sleep(restDuration);
    this.isResting = false;
    this.totalWorkTime += (this.workEndTime - this.workStartTime);
    logger.info('✅ Ket thuc nghi ngoi');
  }

  isWorkTimeRemaining() {
    if (!this.workEndTime) return false;
    return Date.now() < this.workEndTime;
  }

  getRemainingWorkTime() {
    if (!this.workEndTime) return 0;
    const remaining = Math.max(0, (this.workEndTime - Date.now()) / 60000);
    return Math.ceil(remaining);
  }

  getStatus() {
    return {
      isResting: this.isResting,
      workStartTime: this.workStartTime,
      workEndTime: this.workEndTime,
      remainingMinutes: this.getRemainingWorkTime(),
      cycleCount: this.cycleCount,
      totalWorkTime: this.totalWorkTime,
      totalRestTime: this.totalRestTime
    };
  }

  reset() {
    this.workStartTime = null;
    this.workEndTime = null;
    this.isResting = false;
    this.cycleCount = 0;
    this.totalWorkTime = 0;
    this.totalRestTime = 0;
  }
}

module.exports = ScheduleManager;`);

const BlockChecker = require('./check_block');

class MainApplication {

  async mainLoop() {

          const waitEnd = Date.now() + cooldown * 60 * 1000;
          while (Date.now() < waitEnd && this.isRunning) {
            const remaining = Math.ceil((waitEnd - Date.now()) / 60000);
            const hours = Math.floor(remaining / 60);
            const minutes = remaining % 60;
            logger.info('⏳ Con ' + hours + 'h' + minutes + ' phut cho');
            await sleep(600000);
          }
          if (!this.isRunning) break;
          logger.info('🔄 Kiem tra lai sau 24 gio...');

          if (isStillBlocked) {
            logger.error('❌ Van bi chan, dung he thong!');
            this.isRunning = false;
            break;
          } else {
            logger.info('✅ Da het chan, tiep tuc hoat dong');
            this.isBlocked = false;
            this.blockChecker.resetCheckCount();
            continue;
          }
        }
        if (!this.isBlocked) {

          while (this.scheduleManager.isWorkTimeRemaining() && this.isRunning) {
