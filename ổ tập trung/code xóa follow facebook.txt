# -*- coding: utf-8 -*-
import os, re, time, random, requests

COOKIE_FILE = "fb_cookie.txt"

def doc_cookie():
    if os.path.exists(COOKIE_FILE):
        with open(COOKIE_FILE, "r") as f: return f.read().strip()
    return ""

def kiem_tra_cookie_live(cookie):
    if not cookie: return False
    try:
        res = requests.get("https://m.facebook.com/profile.php", headers={'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36', 'Cookie': cookie}, timeout=10)
        if "logout" in res.text.lower() or "save" in res.text.lower() or "bạn bè" in res.text: return True
    except: pass
    return False

def quet_danh_sach_goi_y(cookie):
    print("[*] Dang quet danh sach GOI Y KET BAN (Nguoi ban co the biet)...")
    # Sử dụng link m.facebook để quét đồng bộ danh sách gợi ý mới nhất
    url = "https://m.facebook.com/friends/center/suggestions/"
    uids = []
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
        'Cookie': cookie
    }
    try:
        res = requests.get(url, headers=headers, timeout=12)
        # Quét cấu trúc UID người dùng trong mục gợi ý kết bạn
        matches = re.findall(r'uid=(\d+)', res.text) or re.findall(r'id=(\d+)', res.text)
        for uid in matches:
            if uid not in uids and len(uid) > 10:
                uids.append(uid)
    except Exception as e:
        print(f"[-] Loi khi quet goi y: {e}")
    return uids

def gui_loi_moi_ket_ban(uid, cookie):
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36',
        'Cookie': cookie,
        'Referer': f'https://m.facebook.com/{uid}'
    }
    try:
        # Kiểm tra sự tồn tại của nút kết bạn trước khi bấm
        check_res = requests.get(f"https://m.facebook.com/{uid}", headers=headers, timeout=10)
        if "add_friend" not in check_res.text and "a/friendrequest" not in check_res.text and "Thêm bạn bè" not in check_res.text:
            print(f"[-] UID {uid} khong co nut ket ban hoac da gui truoc do. Bo qua!")
            return "skip"
            
        # Gửi gói tin kết bạn lên hệ thống Facebook
        res = requests.post(f"https://m.facebook.com/a/friendrequest/?to_friend={uid}", headers=headers, timeout=10)
        
        # BỘ LỌC QUÉT PHÁT HIỆN KHI BỊ FACEBOOK GIỚI HẠN TÍNH NĂNG
        detect_block_keywords = ["checkpoint", "xc_error", "chặn", "bị hạn chế", "hạn chế tính năng", "block", "spam"]
        if any(keyword in res.text.lower() for keyword in detect_block_keywords) or "checkpoint" in res.url:
            print(f"[!!!] CANH BAO: PHAT HIEN NICK BI FACEBOOK GIOI HAN TINH NANG KET BAN!")
            return "blocked"
            
        if res.status_code == 200 or "đã gửi" in res.text.lower():
            print(f"[+] Da gui loi moi ket ban thanh cong den UID: {uid}")
            return "success"
    except Exception as e:
        print(f"[-] Loi ket noi den UID {uid}: {e}")
    return "fail"

def main():
    print("[*] Tool bat dau vao vong lap VO HAN theo cau hinh:")
    print("[>] Quy trinh: Gui 10 nguoi lien tuc trong 1 tieng -> Nghi 10 phut -> Lap lai.")
    print("[>] Khi Facebook bat gioi han -> Tu dong ngu dong 24 tieng roi chay lai.")
    
    while True:
        cookie = doc_cookie()
        if not kiem_tra_cookie_live(cookie):
            print("[-] Cookie bi DIE hoac khong thay file fb_cookie.txt. Cho 5 phut test lai...")
            time.sleep(300)
            continue

        danh_sach_uid = quet_danh_sach_goi_y(cookie)
        if not danh_sach_uid:
            print("[-] Khong tim thay ai trong danh sach goi y luc nay. Doi 10 phut quet lai...")
            time.sleep(600)
            continue

        print(f"[+] Quet thanh cong. Tim thay {len(danh_sach_uid)} nguoi trong danh sach goi y.");
        so_luong_can_add = min(10, len(danh_sach_uid))
        uids_luot_nay = random.sample(danh_sach_uid, so_luong_can_add)
        
        bi_chan = False
        count_success = 0
        
        print(f"[*] Bat dau tien hanh gui ket ban cho {so_luong_can_add} nguoi...")
        for uid in uids_luot_nay:
            ket_qua = gui_loi_moi_ket_ban(uid, cookie)
            
            if ket_qua == "blocked":
                bi_chan = True
                break # Ngắt ngay lượt chạy này nếu phát hiện dính chặn
                
            if ket_qua == "success":
                count_success += 1
                
            # Khoảng cách nghỉ an toàn ngắn giữa các lần click bấm kết bạn (từ 10 đến 25 giây)
            time.sleep(random.randint(10, 25))

        # XỬ LÝ KHI BỊ CHẶN: TỰ ĐỘNG NGỦ ĐÔNG ĐÚNG 24 TIẾNG
        if bi_chan:
            print("[❌] PHAT HIEN BI CHAN TINH NANG! Dung toan bo tool, ngu dong 24 TIENG...")
            time.sleep(24 * 3600) # Nghỉ đúng 86400 giây
            print("[*] Da qua 24 tieng cach ly. Khoi dong lai vong lap de tiep tuc kiem tra...")
            continue

        print(f"[===>] Hoan thanh dot gui. Da add thanh cong {count_success}/{so_luong_can_add} nguoi.")
        print("[*] Dung luot chay trong 1 tieng. Tien hanh nghi 10 phut truoc khi lap lai...")
        time.sleep(600) # Nghỉ đúng 10 phút (600 giây) theo yêu cầu rồi lặp lại vòng mới

if __name__ == "__main__":
    main()
