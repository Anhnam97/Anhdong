class ScheduleManager {
  constructor() {
    this.settings = getSettings();
    this.workStartTime = null;
    this.workEndTime = null;
    this.isResting = false;
    this.cycleCount = 0;
  }

  startWorkCycle() {
    this.cycleCount++;
    this.workStartTime = Date.now();
    const workDuration = (this.settings.work_duration || 60) * 60 * 1000;
    this.workEndTime = this.workStartTime + workDuration;
    this.isResting = false;
    logger.info('🚀 Bat dau chu ky lam viec #' + this.cycleCount);
  }

  async startRestCycle() {
    this.isResting = true;
    const restDuration = (this.settings.rest_duration || 10) * 60 * 1000;
    logger.info('☕ Nghi ngoi ' + (this.settings.rest_duration || 10) + ' phut');
    await sleep(restDuration);
    this.isResting = false;
    logger.info('✅ Ket thuc nghi ngoi');
  }

  isWorkTimeRemaining() {
    if (!this.workEndTime) return false;
    return Date.now() < this.workEndTime;
  }

  getRemainingWorkTime() {
    if (!this.workEndTime) return 0;
    return Math.ceil(Math.max(0, (this.workEndTime - Date.now()) / 60000));
  }

  getStatus() {
    return {
      isResting: this.isResting,
      remainingMinutes: this.getRemainingWorkTime(),
      cycleCount: this.cycleCount
    };
  }
}

module.exports = ScheduleManager;`);
